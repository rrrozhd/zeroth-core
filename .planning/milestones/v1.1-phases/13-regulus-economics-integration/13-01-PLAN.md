    ```python
    class InstrumentedProviderAdapter:
        """Wraps any ProviderAdapter to emit Regulus cost events per D-04."""

        def __init__(
            self,
            inner: ProviderAdapter,
            regulus_client: RegulusClient,
            cost_estimator: CostEstimator,
            *,
            node_id: str,
            run_id: str,
            tenant_id: str,
            deployment_ref: str,
        ) -> None: ...

        async def ainvoke(self, request: ProviderRequest) -> ProviderResponse:
            # 1. Call inner.ainvoke(request)
            # 2. Estimate cost via CostEstimator.estimate(model_name, input_tokens, output_tokens)
            # 3. Build ExecutionEvent with:
            #    - capability_id=self._node_id (per RESEARCH mapping)
            #    - implementation_id=request.model_name
            #    - model_version=request.model_name
            #    - token_cost_usd=estimated_cost (Decimal)
            #    - latency_ms=elapsed (perf_counter)
            #    - compute_time_ms=elapsed
            #    - metadata={"run_id": ..., "tenant_id": ..., "deployment_ref": ..., "input_tokens": ..., "output_tokens": ..., "total_tokens": ...}
            # 4. Fire-and-forget via regulus_client.track_execution(event)
            # 5. Return response.model_copy(update={"cost_usd": float(cost), "cost_event_id": event.execution_id})
    ```
