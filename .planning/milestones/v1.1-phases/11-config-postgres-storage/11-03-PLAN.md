@pytest.fixture(scope="session")
def postgres_container():
    """Session-scoped Postgres container for integration tests."""
    with PostgresContainer("postgres:17") as pg:
        yield pg

@pytest.fixture
async def postgres_database(postgres_container):
    """Async Postgres database for tests."""
    from zeroth.storage.async_postgres import AsyncPostgresDatabase
    from zeroth.service.bootstrap import run_migrations

    # Get connection URL, convert from psycopg2 format to psycopg3/sqlalchemy format
    url = postgres_container.get_connection_url()
    # testcontainers returns psycopg2 URL; we need postgresql+psycopg for SQLAlchemy
    sa_url = url.replace("psycopg2", "psycopg")
    # For psycopg3 direct connection, strip the driver
    dsn = url.replace("postgresql+psycopg2://", "postgresql://")

    # Run Alembic migrations
    run_migrations(sa_url)

    db = await AsyncPostgresDatabase.create(dsn, min_size=1, max_size=3)
    yield db
    await db.close()

    # Clean tables between tests (truncate all)
    import psycopg
    conn = await psycopg.AsyncConnection.connect(dsn)
    async with conn:
        async with conn.transaction():
            for table in ["audit_records", "approval_records", "runs", "threads",
                          "run_checkpoints", "graph_versions", "contract_versions",
