    Multi-stage build using uv:

    ```dockerfile
    # ============================================================
    # Stage 1: Builder — install all dependencies via uv
    # ============================================================
    FROM python:3.12-slim AS builder

    COPY --from=ghcr.io/astral-sh/uv:latest /uv /uvx /bin/

    # GovernAI requires git for its GitHub commit pin (Pitfall 2)
    RUN apt-get update && apt-get install -y --no-install-recommends git && rm -rf /var/lib/apt/lists/*

    ENV UV_COMPILE_BYTECODE=1
    ENV UV_LINK_MODE=copy

    WORKDIR /app

    # Copy dependency manifests first for layer caching
    COPY pyproject.toml uv.lock ./

    # Copy pre-built Regulus SDK wheel (per D-03, Pitfall 1)
    COPY docker/regulus-sdk/ /wheels/

    # Install dependencies (without the project itself).
    # The Regulus local path won't resolve inside Docker, so we:
    # 1. Install everything except the project
    # 2. Then install the Regulus wheel separately
    RUN uv sync --locked --no-install-project --no-dev || \
        uv sync --no-install-project --no-dev

    # Install the pre-built Regulus SDK wheel explicitly
    RUN uv pip install /wheels/*.whl 2>/dev/null || true

    # Copy source and install the project itself
    COPY src/ src/
    RUN uv sync --locked --no-dev || uv sync --no-dev

    # ============================================================
    # Stage 2: Runtime — minimal image with only what's needed
    # ============================================================
    FROM python:3.12-slim

    WORKDIR /app

    # Non-root user for security
    RUN useradd --create-home --uid 1001 zeroth
    USER zeroth

    # Copy virtual environment and source from builder
    COPY --from=builder --chown=zeroth:zeroth /app/.venv /app/.venv
    COPY --from=builder --chown=zeroth:zeroth /app/src /app/src

    ENV PATH="/app/.venv/bin:$PATH"

    EXPOSE 8000

    # Health check for Docker (uses the /health/live endpoint from Plan 17-01)
    HEALTHCHECK --interval=15s --timeout=5s --start-period=30s --retries=3 \
        CMD python -c "import urllib.request; urllib.request.urlopen('http://localhost:8000/health/live')" || exit 1

    # Run uvicorn with proxy-headers for correct client IP behind Nginx
    # The bootstrap module's create_app handles migration running and app creation.
    # For Docker, we use a factory pattern: uvicorn calls the factory function.
    CMD ["python", "-m", "uvicorn", "zeroth.service.bootstrap:create_app", "--host", "0.0.0.0", "--port", "8000", "--proxy-headers", "--factory"]
    ```

    **IMPORTANT on CMD:** The bootstrap module has `bootstrap_app` which is async and requires database + deployment_ref. For Docker, we need a simpler entrypoint. Create a thin `src/zeroth/service/entrypoint.py` module:

    ```python
    """Docker entrypoint for the Zeroth platform."""
    import asyncio
    from zeroth.config.settings import get_settings
    from zeroth.storage import create_database
    from zeroth.service.bootstrap import bootstrap_app, run_migrations

    async def _create_app():
        settings = get_settings()
        # Run migrations
        if settings.database.backend == "postgres" and settings.database.postgres_dsn:
            run_migrations(settings.database.postgres_dsn.get_secret_value())
        db = await create_database(settings)
        return await bootstrap_app(db, deployment_ref=settings.deployment_ref)

    def create_app():
        return asyncio.run(_create_app())
    ```

    Actually, this is complex and depends on patterns not yet visible. The simpler approach: use a shell script entrypoint or the existing bootstrap pattern. Looking at `bootstrap.py`, there is no `create_app_from_settings` factory yet. The CMD should point to whatever factory function exists.

    **Revised approach:** Rather than creating a new entrypoint module (which requires understanding the full bootstrap flow), the Dockerfile should use a simpler CMD that the user can customize. Use:

    ```dockerfile
    CMD ["python", "-m", "uvicorn", "zeroth.service.app:create_app", "--host", "0.0.0.0", "--port", "8000", "--proxy-headers"]
    ```

    But `create_app` requires a `bootstrap` argument. The production entrypoint pattern needs investigation. For now, create the Dockerfile with a placeholder CMD and document that the user must set the entrypoint command based on their deployment setup. Use an environment-variable-driven approach:

    And create `src/zeroth/service/entrypoint.py`:
    ```python
    """Production entrypoint for Zeroth inside Docker.

    Runs Alembic migrations (if Postgres backend), bootstraps the service,
    and starts uvicorn.
    """
    from __future__ import annotations

    import asyncio
    import os
    import sys

    import uvicorn


    def main() -> None:
        """Run migrations and start the Zeroth platform."""
        # Import settings to resolve database config
        from zeroth.config.settings import get_settings

        settings = get_settings()

        # Run Alembic migrations for Postgres backend
        if settings.database.backend == "postgres" and settings.database.postgres_dsn:
            from zeroth.service.bootstrap import run_migrations
            dsn = settings.database.postgres_dsn.get_secret_value()
            print(f"Running Alembic migrations against Postgres...", flush=True)
            run_migrations(dsn)
            print("Migrations complete.", flush=True)

        # Determine TLS settings
        ssl_keyfile = settings.tls.keyfile if hasattr(settings, 'tls') else None
        ssl_certfile = settings.tls.certfile if hasattr(settings, 'tls') else None

        uvicorn.run(
            "zeroth.service.entrypoint:app_factory",
            host="0.0.0.0",
            port=int(os.environ.get("PORT", "8000")),
            proxy_headers=True,
            factory=True,
            ssl_keyfile=ssl_keyfile,
            ssl_certfile=ssl_certfile,
        )


    async def _bootstrap() -> object:
        from zeroth.config.settings import get_settings
        from zeroth.service.bootstrap import bootstrap_service, ServiceBootstrap
        from zeroth.service.app import create_app
        from zeroth.storage.factory import create_async_database

        settings = get_settings()
        database = await create_async_database(settings.database)

        deployment_ref = os.environ.get("ZEROTH_DEPLOYMENT_REF", "default")
        bootstrap = await bootstrap_service(database, deployment_ref=deployment_ref)
        return create_app(bootstrap)


    def app_factory():
        return asyncio.run(_bootstrap())


    if __name__ == "__main__":
        main()
    ```

    **2. Create `docker-compose.yml` (per D-04, D-05, D-16):**

    ```yaml
    services:
      zeroth:
        build: .
        depends_on:
          postgres:
            condition: service_healthy
          redis:
            condition: service_healthy
        environment:
          ZEROTH_DATABASE__BACKEND: postgres
          ZEROTH_DATABASE__POSTGRES_DSN: "postgresql://zeroth:zeroth@postgres:5432/zeroth"
          ZEROTH_REDIS__HOST: redis
          ZEROTH_REDIS__PORT: "6379"
          ZEROTH_SANDBOX__BACKEND: sidecar
          ZEROTH_SANDBOX__SIDECAR_URL: "http://sandbox-sidecar:8001"
          ZEROTH_DISPATCH__ARQ_ENABLED: "true"
          ZEROTH_DEPLOYMENT_REF: "default"
        networks:
          - zeroth-net
        restart: unless-stopped

      postgres:
        image: postgres:16-alpine
        environment:
          POSTGRES_USER: zeroth
          POSTGRES_PASSWORD: zeroth
          POSTGRES_DB: zeroth
        volumes:
          - pgdata:/var/lib/postgresql/data
        healthcheck:
          test: ["CMD-SHELL", "pg_isready -U zeroth"]
          interval: 5s
          timeout: 3s
          retries: 5
        networks:
          - zeroth-net

      redis:
        image: redis:7-alpine
        healthcheck:
          test: ["CMD", "redis-cli", "ping"]
          interval: 5s
          timeout: 3s
          retries: 5
        networks:
          - zeroth-net

      regulus:
        image: regulus-backend:latest
        environment:
          REGULUS_PORT: "8080"
        networks:
          - zeroth-net

      sandbox-sidecar:
        image: zeroth-sandbox-sidecar:latest
        volumes:
          - /var/run/docker.sock:/var/run/docker.sock:ro
        networks:
          - zeroth-net

      nginx:
        image: nginx:1.27-alpine
        ports:
          - "443:443"
          - "80:80"
        volumes:
          - ./docker/nginx/nginx.conf:/etc/nginx/conf.d/default.conf:ro
          - ./docker/nginx/certs:/etc/nginx/certs:ro
        depends_on:
          - zeroth
        networks:
          - zeroth-net
        restart: unless-stopped

    volumes:
      pgdata:

    networks:
      zeroth-net:
        driver: bridge
    ```

    - All connection strings use Docker service names as hostnames
    - regulus and sandbox-sidecar use placeholder images (users build/provide these)

        location / {
            proxy_pass http://zeroth:8000;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
            proxy_buffering off;
            proxy_read_timeout 300s;
            proxy_connect_timeout 10s;
        }
    }

    server {
        listen 80;
        server_name _;
        return 301 https://$host$request_uri;
