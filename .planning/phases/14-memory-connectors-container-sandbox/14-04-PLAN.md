    ```python
    from fastapi import FastAPI, HTTPException
    from zeroth.sandbox_sidecar.models import (
        SidecarExecuteRequest, SidecarExecuteResponse,
        SidecarStatusResponse, SidecarHealthResponse,
    )
    from zeroth.sandbox_sidecar.executor import SidecarExecutor

    app = FastAPI(title="Zeroth Sandbox Sidecar")
    executor = SidecarExecutor()

    @app.post("/execute", response_model=SidecarExecuteResponse)
    async def execute(request: SidecarExecuteRequest) -> SidecarExecuteResponse:
        return await executor.execute(request)

    @app.get("/executions/{execution_id}", response_model=SidecarStatusResponse)
    async def get_status(execution_id: str) -> SidecarStatusResponse:
        result = await executor.get_status(execution_id)
        if result is None:
            raise HTTPException(status_code=404, detail="Execution not found")
        return result

    @app.post("/executions/{execution_id}/cancel")
    async def cancel(execution_id: str) -> dict:
        await executor.cancel(execution_id)
        return {"status": "cancelled"}

    @app.get("/health", response_model=SidecarHealthResponse)
    async def health() -> SidecarHealthResponse:
        available = await executor.check_health()
        return SidecarHealthResponse(docker_available=available)
    ```

    **5. Create `src/zeroth/execution_units/sidecar_client.py` (HTTP client for API container):**
    ```python
    import httpx
    from zeroth.sandbox_sidecar.models import (
        SidecarExecuteRequest, SidecarExecuteResponse,
        SidecarStatusResponse, SidecarHealthResponse,
    )

    class SandboxSidecarClient:
        def __init__(self, base_url: str, *, timeout: float = 60.0) -> None:
            self._client = httpx.AsyncClient(base_url=base_url, timeout=timeout)

        async def execute(self, request: SidecarExecuteRequest) -> SidecarExecuteResponse:
            resp = await self._client.post("/execute", content=request.model_dump_json(), headers={"Content-Type": "application/json"})
            resp.raise_for_status()
            return SidecarExecuteResponse.model_validate_json(resp.content)

        async def get_status(self, execution_id: str) -> SidecarStatusResponse:
            resp = await self._client.get(f"/executions/{execution_id}")
            resp.raise_for_status()
            return SidecarStatusResponse.model_validate_json(resp.content)

        async def cancel(self, execution_id: str) -> None:
            resp = await self._client.post(f"/executions/{execution_id}/cancel")
            resp.raise_for_status()

        async def health(self) -> SidecarHealthResponse:
            resp = await self._client.get("/health")
            resp.raise_for_status()
            return SidecarHealthResponse.model_validate_json(resp.content)

        async def close(self) -> None:
            await self._client.aclose()
    ```

    ```python
    async def _run_via_sidecar(
        self,
        *,
        command: Sequence[str],
        input_text: str | None,
        timeout_seconds: float | None,
        environment: SandboxEnvironment,
        resource_constraints: ResourceConstraints | None = None,
    ) -> SandboxExecutionResult:
        import uuid
        from zeroth.sandbox_sidecar.models import SidecarExecuteRequest

        execution_id = str(uuid.uuid4())
        image_ref = self._docker_image_for(self._config.docker.container_name) if ... else self._config.docker.container_name
        # Build request
        request = SidecarExecuteRequest(
            execution_id=execution_id,
            image=image_ref,
            command=list(command),
            input_text=input_text,
            timeout_seconds=timeout_seconds,
            environment=environment.variables,
            cpu_cores=resource_constraints.cpu_cores if resource_constraints else None,
            memory_mb=resource_constraints.memory_mb if resource_constraints else None,
            max_processes=resource_constraints.max_processes if resource_constraints else None,
            network_access=resource_constraints.network_access if resource_constraints else False,
        )
        response = await self._sidecar_client.execute(request)
        return SandboxExecutionResult(
            command=tuple(command),
            returncode=response.returncode or 1,
            stdout=response.stdout,
            stderr=response.stderr,
            workdir=request.working_directory,
            environment=dict(environment.variables),
            timed_out=response.timed_out,
            duration_seconds=response.duration_seconds,
            cache_key=environment.cache_key,
            backend=SandboxBackendMode.SIDECAR.value,
        )
    ```
