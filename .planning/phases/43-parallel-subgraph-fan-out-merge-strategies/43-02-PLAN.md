Implement the full family of declared fan-in merge strategies (`collect`, `reduce`, `merge`, `custom`) and enforce publish-time validation so invalid strategy/reducer configurations are rejected before DRAFT→PUBLISHED transition. Deliver ORCH-03 and ORCH-04.

Purpose: `ParallelConfig.merge_strategy` literal today is `Literal["collect", "reduce"]` but only `collect` is actually implemented; `reduce` silently falls through.     2. **`src/zeroth/core/parallel/reducers.py`** (NEW FILE) — create with the full module:
       ```python
       """Fan-in reduction strategies for ParallelExecutor.

       Strategies (D-04 literal):
         - collect: return outputs as-is (list preserving branch order, None preserved)
         - merge:   shallow dict merge in branch-index order via dict.update (D-02)
         - reduce:  sequential left-to-right fold with BUILT-IN DEFAULT reducer
                    (_default_fold, last-wins) — no reducer_ref required (D-01 + D-04)
         - custom:  sequential left-to-right fold with USER-SUPPLIED reducer resolved
                    from reducer_ref via importlib (requires reducer_ref per D-04)

       Reducer contract (D-01, research Q7):
           ReducerFn = Callable[[dict[str, Any], dict[str, Any]], dict[str, Any]]
           Receives (accumulator, next_value); returns new accumulator.
           Initial accumulator = first non-None branch output in index order.
           2-arg only — no 3-arg (acc, value, branch_index) overload.
       """
       from __future__ import annotations

       import importlib
       import re
       from collections.abc import Callable
       from typing import Any

       from zeroth.core.parallel.errors import (
           MergeStrategyError,
           ReducerRefValidationError,
       )

       ReducerFn = Callable[[Any, Any], Any]
       StrategyFn = Callable[..., Any]

       # D-16 + threat mitigation: only dotted identifiers like `pkg.mod.fn` or
       # `pkg.mod.sub.fn`. At least one dot required so single-segment modules
       # like "os" are rejected (prevents accidental top-level stdlib imports).
       _REDUCER_REF_PATTERN = re.compile(
           r"^[A-Za-z_][A-Za-z0-9_]*(\.[A-Za-z_][A-Za-z0-9_]*)+$"
       )


       def _reduce_collect(
           outputs: list[dict[str, Any] | None],
       ) -> list[dict[str, Any] | None]:
           """Return outputs unchanged — branch-index order preserved, None preserved."""
           return list(outputs)


       def _reduce_merge(
           outputs: list[dict[str, Any] | None],
       ) -> dict[str, Any]:
           """Shallow dict merge in branch-index order; later branches overwrite earlier (D-02)."""
           merged: dict[str, Any] = {}
           for i, out in enumerate(outputs):
               if out is None:
                   continue   # D-19: failed/null branches don't contribute
               if not isinstance(out, dict):
                   raise MergeStrategyError(
                       f"merge strategy requires dict outputs, branch {i} produced "
                       f"{type(out).__name__}"
                   )
               merged.update(out)
           return merged


       def _reduce_fold(
           outputs: list[dict[str, Any] | None],
           *,
           reducer: ReducerFn,
       ) -> Any:
           """Sequential left-to-right fold with user-supplied 2-arg reducer (D-01).

           Skips None branches per D-19. Returns None if all branches are None.
           Returns the single element if only one non-None branch.
           """
           non_null = [o for o in outputs if o is not None]
           if not non_null:
               return None
           if len(non_null) == 1:
               return non_null[0]
           acc = non_null[0]
           for nxt in non_null[1:]:
               acc = reducer(acc, nxt)
           return acc


       def resolve_reducer_ref(reducer_ref: str) -> ReducerFn:
           """Resolve a dotted import path to a callable.

           Security: only strings matching _REDUCER_REF_PATTERN are passed to
           importlib — this rejects paths with spaces, colons, or single-segment
           module names BEFORE any code runs. Graph authors are trusted (graph
           publish requires authenticated API access) — this is defense in depth.

           Raises ReducerRefValidationError for any failure.
           """
           if not isinstance(reducer_ref, str) or not _REDUCER_REF_PATTERN.match(reducer_ref):
               raise ReducerRefValidationError(
                   f"reducer_ref '{reducer_ref}' is not a valid dotted import path; "
                   f"expected pattern: module.submodule.function"
               )
           module_path, _, attr = reducer_ref.rpartition(".")
           try:
               module = importlib.import_module(module_path)
           except ImportError as exc:
               raise ReducerRefValidationError(
                   f"reducer module '{module_path}' not importable: {exc}"
               ) from exc
           if not hasattr(module, attr):
               raise ReducerRefValidationError(
                   f"reducer '{attr}' not found in module '{module_path}'"
               )
           fn = getattr(module, attr)
           if not callable(fn):
               raise ReducerRefValidationError(
                   f"reducer_ref '{reducer_ref}' resolved to {type(fn).__name__}, "
                   f"expected a callable"
               )
           return fn


       def _default_fold(acc: Any, nxt: Any) -> Any:
           """Built-in default reducer for `merge_strategy="reduce"` (D-04 literal).

           Trivial last-wins fold: returns the next value unchanged, discarding
           the accumulator. This satisfies D-01 (a sequential left-to-right fold
           from branch 0 exists and runs) and gives `reduce` a distinct identity
           from `custom` (reduce = platform default; custom = user-supplied
           dotted-path reducer). For non-trivial reductions, author must use
           `merge_strategy="custom"` with an explicit `reducer_ref`.

           A named function (not a lambda) is used so tracebacks are traceable.
           """
           return nxt


       _STRATEGY_REGISTRY: dict[str, StrategyFn] = {
           "collect": _reduce_collect,
           "merge": _reduce_merge,
           # "reduce" dispatches to _reduce_fold(outputs, reducer=_default_fold)
           # "custom" dispatches to _reduce_fold(outputs, reducer=resolve_reducer_ref(ref))
       }


       def dispatch_strategy(
           strategy: str,
           outputs: list[dict[str, Any] | None],
           *,
           reducer_ref: str | None = None,
       ) -> Any:
           """Unified entry point for collect_fan_in — dispatches to the right handler.

           Per D-04 literal:
             - `reduce` uses `_default_fold` when `reducer_ref is None` (the normal case).
                A non-None `reducer_ref` with `reduce` is rejected by the ParallelConfig
                model validator before reaching this dispatch, so we can assume None.
             - `custom` always requires a non-None `reducer_ref`; resolve it via
                `resolve_reducer_ref`.
           """
           if strategy == "reduce":
               # Model validator guarantees reducer_ref is None here.
               return _reduce_fold(outputs, reducer=_default_fold)
           if strategy == "custom":
               if reducer_ref is None:
                   # Defense in depth — model validator should have caught this.
                   raise MergeStrategyError(
                       "merge_strategy='custom' requires reducer_ref"
                   )
               reducer = resolve_reducer_ref(reducer_ref)
               return _reduce_fold(outputs, reducer=reducer)
           handler = _STRATEGY_REGISTRY.get(strategy)
           if handler is None:
               raise MergeStrategyError(f"unknown merge_strategy '{strategy}'")
           return handler(outputs)
       ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Graph publish API → runtime | Authenticated graph authors submit graph JSON containing `ParallelConfig.reducer_ref`. ## Signature-mismatch mitigation

`importlib`-resolved reducer callables are invoked at runtime with `reducer(acc, next_value)`. If the callable has the wrong arity, Python raises `TypeError` at call time. `_reduce_fold` does NOT inspect signatures (research Q7: signature is trusted).
